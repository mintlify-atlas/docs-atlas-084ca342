---
title: "Firestore security rules"
description: "How the Firestore security model works and how to deploy the rules"
---

The Firestore security rules enforce that users can only read and modify their own data. Every document in the `snippets` and `folders` collections is owned by a single authenticated user, and all read and write operations verify ownership before proceeding.

<Warning>
Do not modify the security rules without testing them first. A misconfiguration could expose other users' data or prevent the app from writing documents entirely.
</Warning>

## Security model

The rules use three helper functions to keep permission logic readable and consistent.

### isAuthenticated()

Checks that `request.auth` is not null — meaning the request comes from a signed-in user.

```javascript
function isAuthenticated() {
  return request.auth != null;
}
```

### isOwner(ownerId)

Checks that the authenticated user's UID matches the `ownerId` stored on the document. This ensures users can only modify their own records.

```javascript
function isOwner(ownerId) {
  return isAuthenticated() && request.auth.uid == ownerId;
}
```

### isAdmin()

A special developer-only check based on a verified email address. This grants the developer account full read and write access across all documents, which is useful for debugging and data maintenance.

```javascript
function isAdmin() {
  return isAuthenticated() &&
    (request.auth.token.email == "cmcrossno@gmail.com" && request.auth.token.email_verified == true);
}
```

<Note>
The admin check is email-based and intended for the project owner's account. If you fork or self-host this project, update this email address to your own before deploying.
</Note>

## Snippet validation

Before allowing a create or update operation on a snippet, the rules run `isValidSnippet()` against the incoming data.

**Required fields** — the document must include all of these:

| Field | Type | Constraints |
| --- | --- | --- |
| `title` | string | 1–100 characters |
| `code` | string | 1–10,000 characters |
| `ownerId` | string | Must equal the authenticated user's UID |
| `createdAt` | string | ISO 8601 timestamp |
| `updatedAt` | string | ISO 8601 timestamp |

**Optional fields** — these are permitted but not required:

| Field | Type | Constraints |
| --- | --- | --- |
| `language` | string | — |
| `description` | string | — |
| `tags` | list | Must be a list if present |
| `isFavorite` | boolean | Must be a boolean if present |
| `isPinned` | boolean | Must be a boolean if present |
| `lastCopiedAt` | string | ISO 8601 timestamp |
| `folderId` | string \| null | Must be a string or null if present |

The rules also enforce a strict field whitelist — any document containing a field not in the allowed list is rejected.

## Folder validation

Before allowing a create or update operation on a folder, the rules run `isValidFolder()` against the incoming data.

**Required fields** — the document must include all of these, and no additional fields are permitted:

| Field | Type | Constraints |
| --- | --- | --- |
| `name` | string | 1–50 characters |
| `ownerId` | string | Must equal the authenticated user's UID |
| `createdAt` | string | ISO 8601 timestamp |
| `updatedAt` | string | ISO 8601 timestamp |

## CRUD permissions

The following table summarizes who can perform each operation on snippets and folders.

| Operation | Condition |
| --- | --- |
| **read** | Authenticated + owner of the document (or admin) |
| **create** | Authenticated + data passes validation (or admin) |
| **update** | Authenticated + owner + data passes validation (or admin) |
| **delete** | Authenticated + owner (or admin) |

<Note>
The `/users/{userId}` collection grants full read and write access to the document owner (`request.auth.uid == userId`). This collection stores per-user preferences and settings.
</Note>

## Deploying the rules

<Steps>
  <Step title="Install the Firebase CLI">
    If you haven't already, install the Firebase CLI:

    ```bash
    npm install -g firebase-tools
    ```
  </Step>
  <Step title="Authenticate and select your project">
    ```bash
    firebase login
    firebase use your_project_id
    ```
  </Step>
  <Step title="Deploy only the Firestore rules">
    ```bash
    firebase deploy --only firestore:rules
    ```

    This deploys the `firestore.rules` file without touching any other Firebase resources.
  </Step>
</Steps>

## Full rules file

```javascript firestore.rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Collection: snippets
    // Fields: title, code, language, description, tags, isFavorite, isPinned, ownerId, lastCopiedAt, createdAt, updatedAt, folderId
    // Collection: folders
    // Fields: name, ownerId, createdAt, updatedAt

    function isAuthenticated() {
      return request.auth != null;
    }

    function isOwner(ownerId) {
      return isAuthenticated() && request.auth.uid == ownerId;
    }

    function isAdmin() {
      return isAuthenticated() &&
        (request.auth.token.email == "cmcrossno@gmail.com" && request.auth.token.email_verified == true);
    }

    function isValidSnippet(data) {
      let requiredFields = ['title', 'code', 'ownerId', 'createdAt', 'updatedAt'];
      let allowedFields = ['title', 'code', 'ownerId', 'createdAt', 'updatedAt', 'language', 'description', 'tags', 'isFavorite', 'isPinned', 'lastCopiedAt', 'folderId'];
      
      return data.keys().hasAll(requiredFields) &&
             data.keys().hasOnly(allowedFields) &&
             data.title is string && data.title.size() > 0 && data.title.size() <= 100 &&
             data.code is string && data.code.size() > 0 && data.code.size() <= 10000 &&
             data.ownerId == request.auth.uid &&
             (!('language' in data) || data.language is string) &&
             (!('description' in data) || data.description is string) &&
             (!('tags' in data) || data.tags is list) &&
             (!('isFavorite' in data) || data.isFavorite is bool) &&
             (!('isPinned' in data) || data.isPinned is bool) &&
             (!('folderId' in data) || data.folderId is string || data.folderId == null);
    }

    function isValidFolder(data) {
      let requiredFields = ['name', 'ownerId', 'createdAt', 'updatedAt'];
      let allowedFields = ['name', 'ownerId', 'createdAt', 'updatedAt'];
      
      return data.keys().hasAll(requiredFields) &&
             data.keys().hasOnly(allowedFields) &&
             data.name is string && data.name.size() > 0 && data.name.size() <= 50 &&
             data.ownerId == request.auth.uid;
    }

    match /snippets/{snippetId} {
      allow read: if isAuthenticated() && (resource.data.ownerId == request.auth.uid || isAdmin());
      allow create: if isAuthenticated() && (isValidSnippet(request.resource.data) || isAdmin());
      allow update: if isAuthenticated() && (isOwner(resource.data.ownerId) || isAdmin()) && isValidSnippet(request.resource.data);
      allow delete: if isAuthenticated() && (isOwner(resource.data.ownerId) || isAdmin());
    }

    match /folders/{folderId} {
      allow read: if isAuthenticated() && (resource.data.ownerId == request.auth.uid || isAdmin());
      allow create: if isAuthenticated() && (isValidFolder(request.resource.data) || isAdmin());
      allow update: if isAuthenticated() && (isOwner(resource.data.ownerId) || isAdmin()) && isValidFolder(request.resource.data);
      allow delete: if isAuthenticated() && (isOwner(resource.data.ownerId) || isAdmin());
    }

    match /users/{userId} {
      allow read, write: if isAuthenticated() && request.auth.uid == userId;
    }
  }
}
```
